Caesars Cipher

      
const rot13 = str => {
  let strArr = str.split("")
  let dArr = [];
  strArr.map(c => {
    let asciiCode = c.charCodeAt(0)

    if (asciiCode < 65 || asciiCode > 90) {
      dArr.push(String.fromCharCode(asciiCode))
    }
    else if (asciiCode <= 77) {
      dArr.push(String.fromCharCode(asciiCode + 13))
    }
    else {
      dArr.push(String.fromCharCode(asciiCode - 13))
    }
  })

  let dString = dArr.join('');

  return dString;
}

rot13("SERR PBQR PNZC");
