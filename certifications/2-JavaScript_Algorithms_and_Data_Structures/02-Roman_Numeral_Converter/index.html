Roman Numeral Converter

      
const dec = [1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1];
const rom = ["M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I"];

const convertToRoman = num => {
  let rs = "";
  
  for (var i = 0; i < dec.length; i++) {
    while (dec[i] <= num) {
      rs += rom[i];
      num -= dec[i];
    }
  }

  return rs;
};

convertToRoman(36); // XXXVI
