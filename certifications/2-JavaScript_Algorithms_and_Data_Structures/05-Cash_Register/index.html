Cash Register

      
const val = [
  { name: "ONE HUNDRED", val: 100.0 },
  { name: "TWENTY", val: 20.0 },
  { name: "TEN", val: 10.0 },
  { name: "FIVE", val: 5.0 },
  { name: "ONE", val: 1.0 },
  { name: "QUARTER", val: 0.25 },
  { name: "DIME", val: 0.1 },
  { name: "NICKEL", val: 0.05 },
  { name: "PENNY", val: 0.01 }
];

const checkCashRegister = (price, cash, cid) => {
  let res = { status: null, change: [] };
  let change = cash - price;

  let cashDrawer = cid.reduce(
    (a, c) => {
      a.total += c[1];
      a[c[0]] = c[1];
      return a;
    },
    { total: 0 }
  );

  if (cashDrawer.total === change) {
    res.status = "CLOSED";
    res.change = cid;
    return res;
  }

  if (cashDrawer.total < change) {
    res.status = "INSUFFICIENT_FUNDS";
    return res;
  }

  let changeArr = val.reduce((a, c) => {
    let val = 0;

    while (cashDrawer[c.name] > 0 && change >= c.val) {
      change -= c.val;
      cashDrawer[c.name] -= c.val;
      val += c.val;

      change = Math.round(change * 100) / 100;
    }
    if (val > 0) {
      a.push([c.name, val]);
    }
    return a;
  }, []);

  if (changeArr.length < 1 || change > 0) {
    res.status = "INSUFFICIENT_FUNDS";
    return res;
  }

  res.status = "OPEN";
  res.change = changeArr;
  return res;
}
